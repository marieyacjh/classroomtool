// Refactored Classroom Duel Board Code const CONFIG = { // Constants MAX_STUDENTS: 10, TIMER_INTERVAL: 1000 }; const appState = { students: [], duelMode: false, }; const elements = { timerDisplay: document.getElementById('timerDisplay'), studentList: document.getElementById('studentList'), duelButton: document.getElementById('duelButton'), }; function createStudentRowElement(student) { const row = document.createElement('tr'); row.innerHTML = `${student.name}${student.score}`; return row; } function formatTime(seconds) { const minutes = Math.floor(seconds / 60); const secs = seconds % 60; return `${minutes}:${secs < 10 ? '0' : ''}${secs}`; } function startTimer() { let time = 0; setInterval(() => { time += 1; elements.timerDisplay.innerText = formatTime(time); }, CONFIG.TIMER_INTERVAL); } function setupCanvas() { const canvas = document.getElementById('canvas'); const context = canvas.getContext('2d'); context.fillStyle = '#FFFFFF'; context.fillRect(0, 0, canvas.width, canvas.height); } function startDuelMode() { appState.duelMode = true; // Duel mode logic } function updateStudentList() { elements.studentList.innerHTML = ''; appState.students.forEach(student => { elements.studentList.appendChild(createStudentRowElement(student)); }); } elements.duelButton.addEventListener('click', startDuelMode); // Initialize setupCanvas(); startTimer(); updateStudentList();
